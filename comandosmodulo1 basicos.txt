practica 2 (ip staticas y configuraciones para la conectividad)

ip a

sudo nano /etc/network/interface
	auto eth0 < -- con uno ya basta no es necesario 2 ya que esta es tu targeta de red 
	#iface eth0 inet dhcp <-- esto siempre desactivado si lo quieres statico

	auto eth0
	iface eth0 inet static
    	address 192.168.1.50
    	netmask 255.255.255.0
    	gateway 192.168.1.1
    	dns-nameservers 8.8.8.8 8.8.4.4

(sudo systemctl restart networking

sudo ip addr flush dev eth0

practica 3 (como agregar usuarios y grupos)

sudo usermod -aG sudo nombre_de_usuario
sudo groupadd nombre_del_grupo
tail/etc/group
sudo groupdel nombre_del_grupo

practica 4 (creacion de directorios archivos y permisos)


mkdir directorio_que_quieras
cd directorio_creado
touch archivo.txt
vim archivo.txt
	i <-- esto es para que puedas insertar codigo
	lo que quieras
	: wq
chmod 700 archivo.txt
chmod 070 archivo.txt
ls -l
cd ..
sudo cp directorio/archivo.txt materia2/
rm -rf directorio

[como funcionan los permisos]
	r (read) – 4
	w (write) – 2
	x (execute) – 1
