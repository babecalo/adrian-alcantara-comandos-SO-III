ORIGEN="/home/TU_USUARIO/sincronizacion/"
DESTINO="kali@192.168.1.20:/home/kali/sincronizacion/"
LOG="/home/TU_USUARIO/sync.log"

DATE=$(date '+%Y-%m-%d %H:%M:%S')
echo "[$DATE] Iniciando sync..." >> $LOG
rsync -avz -e "ssh -i /home/TU_USUARIO/.ssh/id_rsa -o StrictHostKeyChecking=no" $ORIGEN $DESTINO >> $LOG 2>&1
echo "[$DATE] Sync completado." >> $LOG

# Dar permisos
chmod +x /home/TU_USUARIO/sync.sh

# Probar manualmente
bash /home/TU_USUARIO/sync.sh

# Ver qué pasó
cat /home/TU_USUARIO/sync.log

crontab -e
	* * * * * /bin/bash /home/TU_USUARIO/sync.sh

# Ver logs del sistema para confirmar que crontab corre
grep CRON /var/log/syslog | tail -20

# O en tiempo real
sudo tail -f /var/log/syslog | grep CRON

		Practica 2

sudo nano /etc/hosts

# Agregar:
192.168.1.10   node1
192.168.1.20   node2

sudo apt update
sudo apt install pacemaker corosync pcs -y

# Establecer contraseña para usuario hacluster
sudo passwd hacluster
# (usa la misma contraseña en ambos nodos, ej: Cluster123)

# Habilitar servicios
sudo systemctl enable pcsd && sudo systemctl start pcsd

# Autenticar nodos
sudo pcs host auth node1 node2 -u hacluster -p Cluster123

# Crear el cluster
sudo pcs cluster setup MiCluster node1 node2 --force

# Iniciar cluster
sudo pcs cluster start --all
sudo pcs cluster enable --all

# Verificar estado
sudo pcs status

# Deshabilitar STONITH (para laboratorio)
sudo pcs property set stonith-enabled=false

# Ignorar quorum (solo 2 nodos)
sudo pcs property set no-quorum-policy=ignore

# Verificar
sudo pcs property list

sudo pcs resource create VirtualIP ocf:heartbeat:IPaddr2 \
    ip=192.168.1.100 \
    cidr_netmask=24 \
    op monitor interval=30s

# Ver estado
sudo pcs status
sudo pcs resource status

# En otra PC - hacer ping continuo a la IP flotante
ping 192.168.1.100

# Mientras hace ping, apagar node1:
sudo shutdown now

# El ping debe continuar porque node2 toma la IP flotante
# Volver a encender node1 y ver que el cluster se recupera
sudo pcs status

sudo nano /etc/keepalived/keepalived.conf

vrrp_instance VI_1 {
    state MASTER
    interface eth0          # Cambia por tu interfaz (ip a para verla)
    virtual_router_id 51
    priority 100            # Master tiene prioridad más alta
    advert_int 1

    authentication {
        auth_type PASS
        auth_pass 1234
    }

    virtual_ipaddress {
        192.168.1.100/24    # Tu IP flotante
    }
}

sudo nano /etc/keepalived/keepalived.conf

vrrp_instance VI_1 {
    state BACKUP
    interface eth0          # Cambia por tu interfaz
    virtual_router_id 51
    priority 90             # Backup tiene prioridad más baja
    advert_int 1

    authentication {
        auth_type PASS
        auth_pass 1234
    }

    virtual_ipaddress {
        192.168.1.100/24
    }
}
