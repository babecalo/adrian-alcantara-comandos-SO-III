	smtp_use_tls = yes
	smtp_sasl_auth_enable = yes
	smtp_sasl_password_maps = hast:/etc/postfix/sasl_passwd
	smtp_sasl_security_options = noanonymous
(ctrl + o)
(ctrl + x)

	[smtp.gmail.com]:587 tucorreo@gmail.com:tu_password_de_app
(ctrl + o)
(ctrl + x)

sudo postmap /etc/postfix/sasl_passwd
sudo chmod 0600 /etc/postfix/sasl_passwd /etc/postfix/sasl_passwd.db

http://localhost:631
